    <script>
        function simulateScan() {
            // Hide QR section and show farmer profile
            document.querySelector('.qr-section').style.display = 'none';
            document.querySelector('.farmer-profile').style.display = 'block';
            
            // Smooth scroll to farmer profile
            document.getElementById('farmer-profile').scrollIntoView({ 
                behavior: 'smooth' 
            });
            
            // Add a subtle animation effect
            setTimeout(() => {
                document.querySelector('.profile-card').style.animation = 'fadeInUp 0.8s ease';
            }, 500);
        }

        // Smooth scrolling for navigation links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });

        // Header scroll effect
        window.addEventListener('scroll', () => {
            const header = document.querySelector('header');
            if (window.scrollY > 100) {
                header.style.background = 'rgba(255, 255, 255, 0.98)';
                header.style.boxShadow = '0 10px 40px rgba(31, 38, 135, 0.2)';
            } else {
                header.style.background = 'rgba(255, 255, 255, 0.95)';
                header.style.boxShadow = '0 8px 32px rgba(31, 38, 135, 0.37)';
            }
        });
    </script>
